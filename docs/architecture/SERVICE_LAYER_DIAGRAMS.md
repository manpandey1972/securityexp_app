## ğŸ“‹ Service Registration Details

### Registration Order in service_locator.dart

```dart
Future<void> setupServiceLocator() async {
  // Step 1: Firebase Singletons
  getIt.registerSingleton<FirebaseAuth>(FirebaseAuth.instance);
  getIt.registerSingleton<FirebaseFirestore>(FirebaseFirestore.instance);
  getIt.registerSingleton<FirebaseStorage>(FirebaseStorage.instance);

  // Step 2: Local Storage
  final prefs = await SharedPreferences.getInstance();
  getIt.registerSingleton<SharedPreferences>(prefs);

  // Step 3: Validators (No dependencies)
  getIt.registerSingleton<EmailValidator>(EmailValidator());
  getIt.registerSingleton<PhoneValidator>(PhoneValidator());
  getIt.registerSingleton<MessageValidator>(MessageValidator());
  getIt.registerSingleton<InputSanitizer>(InputSanitizer());

  // Step 4: Data Sources
  getIt.registerLazySingleton<FirestoreChatService>(
    () => FirestoreChatService(firestore: getIt<FirebaseFirestore>()),
  );
  getIt.registerLazySingleton<FirestoreUserService>(
    () => FirestoreUserService(firestore: getIt<FirebaseFirestore>()),
  );

  // Step 5: Repositories (depend on data sources)
  getIt.registerLazySingleton<ChatRepository>(
    () => ChatRepositoryImpl(service: getIt<FirestoreChatService>()),
  );
  getIt.registerLazySingleton<UserRepository>(
    () => UserRepositoryImpl(service: getIt<FirestoreUserService>()),
  );

  // Step 6: Services (depend on repositories)
  getIt.registerLazySingleton<ChatService>(
    () => ChatService(repository: getIt<ChatRepository>()),
  );
  getIt.registerLazySingleton<UserProfileService>(
    () => UserProfileService(repository: getIt<UserRepository>()),
  );

  // Step 7: High-level Services
  getIt.registerLazySingleton<NotificationService>(
    () => NotificationService(),
  );
}
```
